103  assert(this->size() > 1);
104  std::complex<SMARTDSP_FLOAT_TYPE> meanVal = this->mean();
105  std::complex<SMARTDSP_FLOAT_TYPE> sum = 0;
106  for (unsigned i=0; i<this->size(); i++) {
