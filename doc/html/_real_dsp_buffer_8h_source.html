97  assert(this->size() > 0);
98  std::vector<T> scratchBuf = this->buf;
99  std::sort(scratchBuf.begin(), scratchBuf.end());
100  if (this->size() & 1) {
101  // Odd number of samples
102  return scratchBuf[this->size()/2];
103  }
104  else {
105  // Even number of samples. Average the two in the middle.
106  unsigned topHalfIndex = this->size()/2;
107  return (scratchBuf[topHalfIndex] + scratchBuf[topHalfIndex-1]) / ((T) 2);
108  }
