330  assert(this->size() > 0);
331  std::vector<T> tempScratch;
332  std::vector<T> *scratch;
333 
334  if (this->scratchBuf == NULL) {
335  scratch = &tempScratch;

338  scratch = this->scratchBuf;
339  }
340  *scratch = this->buf;
341  std::sort(scratch->begin(), scratch->end());
342 
343  T modeVal = 0;
344  unsigned modeLen = 0;
345  T currentVal = (*scratch)[0];
346  unsigned currentLen = 1;
347 
348  for (unsigned i=1; i<this->size(); i++) {
349  if ((*scratch)[i] == currentVal) {
350  currentLen++;
351  }
352  else {
353  if (currentLen > modeLen) {
354  modeVal = currentVal;
355  modeLen = currentLen;
356  }
357  currentVal = (*scratch)[i];
358  currentLen = 1;
359  }
360  }

364  }
